## 题目
给一个 `n` 求：

$$ \sum_{i=1}^{n} \sum_{j=1}^{n} \left[ \gcd(i, j) = 1 \right] $$

## 代码

等价于求：

$$ \left( 2 \sum_{i=1}^{n} \phi_i \right) + 1 $$

```cpp
#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
const int MAX_N = 50000;
typedef unsigned long long ULL;
int iscomps[MAX_N];
int prime[MAX_N]; int ppos;
ULL phi[MAX_N];


void gen_prime(int up) {
    phi[1] = 1;
    for(int i=2;i<up;i++) {
        if(!iscomps[i]) {
            prime[ppos++] = i;
            phi[i] = i - 1;
        }

        for(int j=0;j<ppos&&i*prime[j]<up;j++) {
            iscomps[i * prime[j]] = 1;
            if(i % prime[j] == 0) {
                phi[i * prime[j]] = phi[i] * prime[j];
                break;
            } else {
                phi[i * prime[j]] = phi[i] * (prime[j] - 1);
            }
        }
    }
}

int main() {
    gen_prime(MAX_N);

    int q; scanf("%d", &q);
    ULL ans = 0;
    for(int i=1;i<q;i++) {
        ans += phi[i];
    }

    ans = ans * 2 + 1;
    printf("%lld\n", ans);
}
```
